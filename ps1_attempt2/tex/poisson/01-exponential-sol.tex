\begin{answer}
\begin{align*}
p(y; \lambda) = \frac{e^{-\lambda} \lambda^y}{y!} = (1/y!) \exp(y \log \lambda - \lambda)\\
\text{i.e.} \quad T(y)=y;\; \eta = \log \lambda ;\; a(\eta) = e^\eta ;\; b(y) = (1/y!)\\
\text{in the form} \quad p(y;\lambda) = p(y;\eta) = b(y) \exp(\eta^T T(y) = a(\eta))\\
\end{align*}
\end{answer}
